% Bibliothèque commune de fonctions de formattage
%
% Inspirées du livret d'exemple de complies bénédictines de Gregorio
%

\newdimen\colpslatin
\setlength{\colpslatin}{\textwidth * \real{0.53}}

\newdimen\colpsvern
\setlength{\colpsvern}{\textwidth * \real{0.43}}

\newdimen\collectlatin
\setlength{\collectlatin}{\textwidth * \real{0.48}}

\newdimen\collectvern
\setlength{\collectvern}{\textwidth * \real{0.48}}

\def\latin#1{%
\selectlanguage{latin}
\ParallelLText{#1}%
\selectlanguage{francais}%
\relax %
}

\def\vern#1{%
\ParallelRText{#1}%
\ParallelPar %
%\kern -1mm%
\relax %
}

\def\firstlatin#1#2#3#4{%
\latin{%
\noindent\begin{minipage}{\ParallelLWidth}%
\grelettrine{#2}{#3}{#1}#4\vspace{0.9mm}\vspace{0.13cm}%
\end{minipage}%
}%
\vspace{-0.2cm}%
}

\def\firstlatinq#1#2#3#4{%
\latin{%
\noindent\begin{minipage}{\ParallelLWidth}%
\grelettrineq{#2}{#3}{#1}#4\vspace{0.9mm}\vspace{0.13cm}%
\end{minipage}%
}%
\vspace{-0.2cm}%
}

\def\twocolspace#1{%
{%
\kern -2pt%
\fontsize{1}{#1}\selectfont%
\latin{~}\vern{~}%
\kern -2pt%
}%
}

\def\twocolstrictspace#1{%
\temp=#1%
\advance\temp by 1mm%
\ParallelLText{\ \vspace*{\temp}}\ParallelRText{\ \vspace*{\temp}}%
\kern -1mm%
\relax %
}

\def\titreantienne#1{%
\begin{center}%
\Large{#1}%
\end{center}%
}

\def\titrerepons#1{%
\begin{center}%
\large{\uppercase{#1}}%
\end{center}%
}

\def\titrelecture#1#2{%
{
%\begin{center}%

\nobreak%
\fontsize{10.5}{12.5}\selectfont%
\centerline{#2}\par%
%\end{center}%
}
}

\def\titrenocturne#1{%

}

\def\titrelaudes#1{%

\def\rubrique#1{%
{%
\parskip=6pt\parfillskip=6pt%
\fontsize{11.5}{12.5}\selectfont%
\textcolor{red}{\itshape #1}%
}%
}

\def\rubriqueBlack#1{%
{\textcolor{black}{\textnormal{#1}}}%
}

Après la reprise de l’antienne, tous se lèvent et,
tournés vers l’autel, répondent au verset proclamé par le chantre.}%
}

reportera aux annexes, page} \pageref{#1}.%
}
\nobreak
}
