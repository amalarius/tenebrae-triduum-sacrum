% Bibliothèque commune de fonctions de formattage
%
% Inspirées du livret d'exemple de complies bénédictines de Gregorio
%

\newdimen\colwidth
\colwidth=5.7cm

\def\latin#1{%
\selectlanguage{latin}
\ParallelLText{#1}%
\selectlanguage{francais}%
\relax %
}

\def\vern#1{%
\ParallelRText{#1}%
\ParallelPar %
%\kern -1mm%
\relax %
}

\def\firstlatin#1#2#3#4{%
\latin{%
\noindent\begin{minipage}{\ParallelLWidth}%
\grelettrine{#2}{#3}{#1}#4\vspace{0.9mm}\vspace{0.13cm}%
\end{minipage}%
}%
\vspace{-0.2cm}%
}

\def\firstlatinq#1#2#3#4{%
\latin{%
\noindent\begin{minipage}{\ParallelLWidth}%
\grelettrineq{#2}{#3}{#1}#4\vspace{0.9mm}\vspace{0.13cm}%
\end{minipage}%
}%
\vspace{-0.2cm}%
}

\def\titreantienne#1{%
\begin{center}%
#1%
\end{center}%
}

\def\titrerepons#1{%
\begin{center}%
#1%
\end{center}%
}

\def\titrelecture#1{%
\begin{center}%
\uppercase{#1}%
\end{center}%
}
